# 基于强化学习的煤矿工作面接续优化研究框架

## 摘要

工作面接续是煤矿生产组织的核心问题，直接影响矿井产能连续性和经济效益。本研究提出一种基于深度强化学习的工作面接续智能优化方法，将接续决策建模为马尔可夫决策过程（MDP），通过智能体与矿井仿真环境的交互学习最优接续策略。该方法能够综合考虑时间衔接、空间约束、设备调度、通风系统等多维度耦合约束，实现接续方案的全局优化。

**关键词**：工作面接续；强化学习；序贯决策；采矿规划；多约束优化

## 1. 研究背景与问题定义

### 1.1 工作面接续问题

工作面接续是指在采区或矿井范围内，合理安排各工作面的开采顺序和时间衔接关系，确保：
- 生产连续性：前一工作面结束时，后续工作面具备开采条件
- 产能均衡：避免产量大幅波动
- 资源优化：设备、人员、系统的高效利用

```python
class Action:
    """接续决策动作"""
    action_type: int           # 动作类型

class ActionType(Enum):
    WAIT = 0                   # 等待（维持当前状态）
    START_PREP = 1             # 开始准备某工作面
    START_MINING = 2           # 开始回采某工作面
    MOVE_EQUIPMENT = 3         # 设备搬家到某工作面
    ACCELERATE_PREP = 4        # 加速准备（投入更多资源）
```

#### 2.3.2 动作空间结构

采用**混合动作空间**设计：

```
动作空间 = {WAIT} ∪ {START_PREP_i} ∪ {START_MINING_i} ∪ {MOVE_EQUIPMENT_i}

其中 i ∈ {可用工作面集合}
```

总动作数：$|\mathcal{A}| = 1 + 3n$（n为工作面数量）

#### 2.3.3 动作掩码（Action Masking）

关键创新：使用动作掩码处理约束，确保只选择合法动作

```python
def get_valid_action_mask(state: MineState) -> np.ndarray:
    """生成合法动作掩码"""
    n_workfaces = len(state.workfaces)
    mask = np.zeros(1 + 3 * n_workfaces, dtype=np.float32)

    # WAIT 总是合法的
    mask[0] = 1.0

    for i, wf in enumerate(state.workfaces):
        # START_PREP: 只有"待准备"状态的工作面
        if wf.status == 0:
            # 检查是否满足前置条件（如相邻工作面状态）
            if _can_start_prep(wf, state):
                mask[1 + i] = 1.0

        # START_MINING: 只有"待采"状态且设备就位
        if wf.status == 2 and wf.equipment_installed:
            if state.equipment_status == 0:  # 设备空闲
                mask[1 + n_workfaces + i] = 1.0

        # MOVE_EQUIPMENT: 设备空闲且目标工作面准备完成
        if wf.status == 2 and not wf.equipment_installed:
            if state.equipment_status == 0:
                mask[1 + 2*n_workfaces + i] = 1.0

    return mask
```

### 2.4 状态转移 $\mathcal{P}$

#### 2.4.1 确定性转移

```python
def step(state: MineState, action: Action) -> MineState:
    """状态转移函数"""
    new_state = deepcopy(state)
    new_state.current_step += 1

    # 1. 执行动作
    if action.action_type == ActionType.START_PREP:
        wf = get_workface(new_state, action.target_workface)
        wf.status = 1  # 准备中
        wf.prep_start_time = new_state.current_step

    elif action.action_type == ActionType.START_MINING:
        wf = get_workface(new_state, action.target_workface)
        wf.status = 3  # 在采
        wf.mining_start_time = new_state.current_step
        new_state.equipment_status = 1  # 生产中

    elif action.action_type == ActionType.MOVE_EQUIPMENT:
        wf = get_workface(new_state, action.target_workface)
        new_state.equipment_status = 2  # 搬家中
        # 计算搬家时间
        move_time = calculate_move_time(state.equipment_location, wf.id)
        schedule_equipment_arrival(new_state, wf.id, move_time)

    # 2. 更新所有工作面状态
    for wf in new_state.workfaces:
        update_workface_progress(wf, new_state)

    # 3. 计算当月产量
    new_state.monthly_production = calculate_monthly_production(new_state)
    new_state.cumulative_production += new_state.monthly_production

    return new_state


def update_workface_progress(wf: WorkfaceState, state: MineState):
    """更新工作面进度"""
    if wf.status == 1:  # 准备中
        # 更新巷道掘进进度
        wf.transport_lane_progress += MONTHLY_TUNNELING_RATE / wf.transport_lane_length
        wf.return_lane_progress += MONTHLY_TUNNELING_RATE / wf.return_lane_length
        wf.cut_progress += MONTHLY_TUNNELING_RATE / wf.cut_length

        # 检查是否准备完成
        if all([wf.transport_lane_progress >= 1.0,
                wf.return_lane_progress >= 1.0,
                wf.cut_progress >= 1.0]):
            wf.status = 2  # 待采

    elif wf.status == 3:  # 在采
        # 更新推进进度
        monthly_advance = calculate_monthly_advance(wf, state)
        wf.current_advance += monthly_advance

        # 检查是否回采完成
        if wf.current_advance >= wf.advance_length:
            wf.status = 4  # 已采
            state.equipment_status = 0  # 设备空闲
            state.equipment_location = wf.id
```

```python
def step_with_stochasticity(state: MineState, action: Action) -> MineState:
    """带随机性的状态转移"""
    new_state = step(state, action)  # 先执行确定性转移

    # 1. 地质条件随机扰动
    for wf in new_state.workfaces:
        if wf.status == 3:  # 在采
            # 实际推进速度 = 计划速度 × 随机系数
            geology_factor = np.random.normal(1.0, 0.1)  # ±10%波动
            wf.current_advance *= geology_factor

    # 2. 设备故障随机事件
    if new_state.equipment_status == 1:  # 生产中

            new_state.equipment_status = 0  # 停机
            new_state.monthly_production *= 0.5  # 产量减半

    # 3. 巷道掘进进度波动
    for wf in new_state.workfaces:
        if wf.status == 1:  # 准备中
            tunneling_factor = np.random.normal(1.0, 0.15)  # ±15%波动
            wf.transport_lane_progress *= tunneling_factor
            wf.return_lane_progress *= tunneling_factor

    return new_state
```

### 2.5 奖励函数 $\mathcal{R}$

#### 2.5.1 奖励设计原则

```
┌─────────────────────────────────────────────────────────────┐
│                      奖励函数设计                            │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  正向奖励 (鼓励)                负向奖励 (惩罚)               │
│  ┌─────────────────┐           ┌─────────────────┐         │
│  │ • 完成月度产量   │           │ • 产量断档       │         │
│  │ • 按时完成接续   │           │ • 设备闲置       │         │
│  │ • 储量高效回收   │           │ • 安全风险       │         │
│  │ • 成本节约      │           │ • 超期准备       │         │
│  └─────────────────┘           └─────────────────┘         │
│                                                             │
│  奖励 = Σ(正向奖励) - Σ(负向惩罚)                            │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

#### 2.5.2 奖励函数实现

```python
def calculate_reward(state: MineState, action: Action,
                     next_state: MineState) -> float:
    """计算即时奖励"""
    reward = 0.0

    # ============ 1. 产量奖励 ============
    production_rate = next_state.monthly_production / next_state.production_target

    if production_rate >= 1.0:
        # 完成目标：正奖励
        reward += REWARD_PRODUCTION_TARGET * min(production_rate, 1.2)
    elif production_rate >= 0.8:
        # 接近目标：小正奖励
        reward += REWARD_PRODUCTION_TARGET * (production_rate - 0.5)
    else:
        # 产量不足：惩罚，且越低惩罚越大
        reward -= PENALTY_LOW_PRODUCTION * (1.0 - production_rate) ** 2

    # ============ 2. 产量连续性奖励 ============
    # 惩罚产量大幅波动
    if state.monthly_production > 0:
        production_change = abs(next_state.monthly_production - state.monthly_production)
        change_rate = production_change / state.monthly_production
        if change_rate > 0.3:  # 波动超过30%
            reward -= PENALTY_PRODUCTION_FLUCTUATION * change_rate

    # ============ 3. 接续及时性奖励 ============
    # 检查是否有工作面即将结束但无后续
    mining_faces = [wf for wf in next_state.workfaces if wf.status == 3]
    ready_faces = [wf for wf in next_state.workfaces if wf.status == 2]

    for mining_wf in mining_faces:
        remaining_months = estimate_remaining_months(mining_wf)
        if remaining_months <= 3:  # 3个月内结束
            if len(ready_faces) == 0:
                # 没有准备好的工作面：严重惩罚
                reward -= PENALTY_NO_SUCCESSION * (4 - remaining_months)
            elif len(ready_faces) >= 1:
                # 有准备好的工作面：奖励
                reward += REWARD_SUCCESSION_READY

    # ============ 4. 设备利用率奖励 ============
    if next_state.equipment_status == 0:  # 设备空闲
        # 检查是否有可开采的工作面
        if any(wf.status == 2 and wf.equipment_installed for wf in next_state.workfaces):
            # 有可采工作面但设备空闲：惩罚
            reward -= PENALTY_EQUIPMENT_IDLE

    # ============ 5. 安全约束奖励 ============
    # 检查相邻工作面同时开采风险
    adjacency_risk = calculate_adjacency_risk(next_state)
    if adjacency_risk > RISK_THRESHOLD:
        reward -= PENALTY_SAFETY_RISK * adjacency_risk

    # ============ 6. 搬家成本惩罚 ============
    if action.action_type == ActionType.MOVE_EQUIPMENT:
        move_distance = calculate_move_distance(state, action.target_workface)
        reward -= PENALTY_MOVE_COST * move_distance / 1000  # 每公里惩罚

    # ============ 7. 终止奖励 ============
    if is_terminal(next_state):
        # 所有工作面开采完成
        total_recovery = next_state.cumulative_production / TOTAL_RESERVES
        reward += REWARD_COMPLETION * total_recovery

        # 额外奖励：提前完成
        if next_state.current_step < PLANNED_DURATION:
            reward += REWARD_EARLY_COMPLETION * (PLANNED_DURATION - next_state.current_step)

    return reward
```

| 算法 | 优点 | 缺点 | 适用性 |

**推荐**：使用 **Masked PPO** 作为主算法，支持动作掩码处理约束。

### 3.2 Masked PPO 算法

#### 3.2.1 网络架构

```python
import torch
import torch.nn as nn

class PolicyNetwork(nn.Module):
    """策略网络：输出动作概率分布"""

    def __init__(self, state_dim: int, action_dim: int, hidden_dim: int = 256):
        super().__init__()

        # 共享特征提取层
        self.feature_extractor = nn.Sequential(
            nn.Linear(state_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
        )

        # 策略头
        self.policy_head = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, action_dim),
        )

        # 价值头
        self.value_head = nn.Sequential(
            nn.Linear(hidden_dim, hidden_dim),
            nn.ReLU(),
            nn.Linear(hidden_dim, 1),
        )

    def forward(self, state: torch.Tensor, action_mask: torch.Tensor):
        features = self.feature_extractor(state)

        # 策略输出
        logits = self.policy_head(features)

        # 应用动作掩码：非法动作设为极小值
        masked_logits = logits.masked_fill(action_mask == 0, -1e9)
        probs = torch.softmax(masked_logits, dim=-1)

        # 价值输出
        value = self.value_head(features)

        return probs, value

    def get_action(self, state: torch.Tensor, action_mask: torch.Tensor):
        probs, value = self.forward(state, action_mask)
        dist = torch.distributions.Categorical(probs)
        action = dist.sample()
        log_prob = dist.log_prob(action)
        return action, log_prob, value
```

```python
class MaskedPPO:
    """带动作掩码的PPO算法"""

    def __init__(self, state_dim, action_dim, config):
        self.policy = PolicyNetwork(state_dim, action_dim)
        self.optimizer = torch.optim.Adam(self.policy.parameters(), lr=config.lr)

        self.clip_epsilon = config.clip_epsilon  # 0.2
        self.gamma = config.gamma                # 0.99
        self.gae_lambda = config.gae_lambda      # 0.95
        self.entropy_coef = config.entropy_coef  # 0.01
        self.value_coef = config.value_coef      # 0.5

    def compute_gae(self, rewards, values, dones):
        """计算广义优势估计 (GAE)"""
        advantages = []
        gae = 0

        for t in reversed(range(len(rewards))):
            if t == len(rewards) - 1:
                next_value = 0
            else:
                next_value = values[t + 1]

            delta = rewards[t] + self.gamma * next_value * (1 - dones[t]) - values[t]
            gae = delta + self.gamma * self.gae_lambda * (1 - dones[t]) * gae
            advantages.insert(0, gae)

        return torch.tensor(advantages)

    def update(self, trajectories):
        """PPO策略更新"""
        states = torch.stack([t.state for t in trajectories])
        actions = torch.tensor([t.action for t in trajectories])
        old_log_probs = torch.tensor([t.log_prob for t in trajectories])
        rewards = [t.reward for t in trajectories]
        dones = [t.done for t in trajectories]
        action_masks = torch.stack([t.action_mask for t in trajectories])

        # 计算旧价值
        with torch.no_grad():
            _, old_values = self.policy(states, action_masks)
            old_values = old_values.squeeze()

        # 计算优势和回报
        advantages = self.compute_gae(rewards, old_values.tolist(), dones)
        returns = advantages + old_values

        # 标准化优势
        advantages = (advantages - advantages.mean()) / (advantages.std() + 1e-8)

        # 多轮更新
        for _ in range(self.n_epochs):
            probs, values = self.policy(states, action_masks)
            dist = torch.distributions.Categorical(probs)

            new_log_probs = dist.log_prob(actions)
            entropy = dist.entropy().mean()

            # 计算比率
            ratio = torch.exp(new_log_probs - old_log_probs)

            # 裁剪目标
            surr1 = ratio * advantages
            surr2 = torch.clamp(ratio, 1 - self.clip_epsilon,
                               1 + self.clip_epsilon) * advantages

            # 损失函数
            policy_loss = -torch.min(surr1, surr2).mean()
            value_loss = nn.MSELoss()(values.squeeze(), returns)

            loss = (policy_loss
                    + self.value_coef * value_loss
                    - self.entropy_coef * entropy)

            self.optimizer.zero_grad()
            loss.backward()
            nn.utils.clip_grad_norm_(self.policy.parameters(), 0.5)
            self.optimizer.step()
```

### 3.3 训练流程

```python
def train(env, agent, config):
    """训练主循环"""

    for episode in range(config.n_episodes):
        state = env.reset()
        trajectories = []
        episode_reward = 0

        while not env.is_terminal():
            # 获取动作掩码
            action_mask = env.get_valid_action_mask()

            # 选择动作
            action, log_prob, value = agent.policy.get_action(
                torch.tensor(state),
                torch.tensor(action_mask)
            )

            # 执行动作
            next_state, reward, done = env.step(action.item())

            # 存储轨迹
            trajectories.append(Trajectory(
                state=torch.tensor(state),
                action=action.item(),
                reward=reward,
                log_prob=log_prob.item(),
                done=done,
                action_mask=torch.tensor(action_mask)
            ))

            state = next_state
            episode_reward += reward

        # 更新策略
        agent.update(trajectories)

        # 日志记录
        if episode % config.log_interval == 0:
            print(f"Episode {episode}, Reward: {episode_reward:.2f}")
            evaluate_and_log(agent, env, episode)
```

### 3.4 算法伪代码

```
算法: 基于Masked PPO的工作面接续优化

输入:
    - 采区工作面集合 W = {w_1, w_2, ..., w_n}
    - 各工作面属性（位置、储量、地质条件）
    - 约束参数（产能目标、安全阈值）

输出:
    - 最优接续策略 π*
    - 接续方案（开采顺序、时间表）

初始化:
    - 策略网络 π_θ，价值网络 V_φ
    - 经验缓冲区 D = ∅
    - 矿井仿真环境 ENV

训练过程:
FOR episode = 1 to N_episodes DO:

    1. 重置环境
       s_0 ← ENV.reset()

    2. 收集轨迹
       FOR t = 0 to T_max DO:
           # 获取合法动作掩码
           mask_t ← GET_VALID_ACTIONS(s_t)

           # 采样动作
           a_t ~ π_θ(·|s_t, mask_t)

           # 执行动作
           s_{t+1}, r_t, done ← ENV.step(a_t)

           # 存储
           D ← D ∪ {(s_t, a_t, r_t, s_{t+1}, mask_t)}

           IF done THEN BREAK
       END FOR

    3. 计算优势估计
       FOR each (s, a, r, s', mask) in D DO:
           δ_t ← r_t + γV_φ(s_{t+1}) - V_φ(s_t)
           A_t ← Σ_{l=0}^{∞} (γλ)^l δ_{t+l}  # GAE
       END FOR

    4. 策略更新 (多轮)
       FOR k = 1 to K DO:
           # 计算概率比
           ratio ← π_θ(a|s,mask) / π_{θ_old}(a|s,mask)

           # 裁剪目标
           L^{CLIP} ← min(ratio·A, clip(ratio,1-ε,1+ε)·A)

           # 价值损失
           L^{VF} ← (V_φ(s) - R_t)^2

           # 熵奖励
           L^{ENT} ← H[π_θ(·|s,mask)]

           # 总损失
           L ← -L^{CLIP} + c_1·L^{VF} - c_2·L^{ENT}

           # 梯度更新
           θ ← θ - α∇_θL
           φ ← φ - α∇_φL
       END FOR

    5. 清空缓冲区
       D ← ∅

END FOR

输出最优策略 π* ← π_θ
```

```python
class MineSuccessionEnv:
    """矿井工作面接续仿真环境"""

    def __init__(self, config: EnvironmentConfig):
        self.config = config
        self.workfaces = self._init_workfaces(config.workface_data)
        self.state = None
        self.step_count = 0

    def _init_workfaces(self, workface_data: List[Dict]) -> List[WorkfaceState]:
        """从设计数据初始化工作面"""
        workfaces = []
        for wf_data in workface_data:
            wf = WorkfaceState(
                id=wf_data['id'],
                status=0,  # 初始状态：待准备
                length=wf_data['length'],
                advance_length=wf_data['width'],
                current_advance=0,
                avg_thickness=wf_data.get('avgThickness', 2.0),
                avg_score=wf_data.get('avgScore', 75),
                reserves=self._calculate_reserves(wf_data),
                transport_lane_progress=0,
                return_lane_progress=0,
                cut_progress=0,
                equipment_installed=False,
            )
            workfaces.append(wf)
        return workfaces

    def reset(self) -> np.ndarray:
        """重置环境"""
        self.step_count = 0

        # 重置所有工作面状态
        for wf in self.workfaces:
            wf.status = 0
            wf.current_advance = 0
            wf.transport_lane_progress = 0
            wf.return_lane_progress = 0
            wf.cut_progress = 0
            wf.equipment_installed = False

        # 初始化全局状态
        self.state = MineState(
            current_step=0,
            workfaces=self.workfaces,
            equipment_location=None,
            equipment_status=0,
            monthly_production=0,
            cumulative_production=0,
            production_target=self.config.monthly_target,
            ventilation_capacity=self.config.ventilation_capacity,
            transport_capacity=self.config.transport_capacity,
            cumulative_cost=0,
            cumulative_revenue=0,
        )

        return encode_state(self.state)

    def step(self, action: int) -> Tuple[np.ndarray, float, bool]:
        """执行一步"""
        # 解析动作
        action_obj = self._decode_action(action)

        # 状态转移
        next_state = self._transition(self.state, action_obj)

        # 计算奖励
        reward = calculate_reward(self.state, action_obj, next_state)

        # 更新状态
        self.state = next_state
        self.step_count += 1

        # 判断终止
        done = self._is_terminal()

        return encode_state(self.state), reward, done

    def _is_terminal(self) -> bool:
        """判断是否终止"""
        # 条件1：所有工作面开采完成
        if all(wf.status == 4 for wf in self.state.workfaces):
            return True

        # 条件2：超过最大时间步
        if self.step_count >= self.config.max_steps:
            return True

        # 条件3：无法继续（陷入死锁）
        if self._is_deadlock():
            return True

        return False
```
